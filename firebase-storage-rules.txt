// Firebase Storage Security Rules for Wificontent App
// Apply these rules in Firebase Console -> Storage -> Rules tab

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    
    // Allow users to upload their own photos to their network folders
    match /users/{uid}/photos/{allPaths=**} {
      allow read: if true;  // Anyone can view photos
      allow write: if true;  // Allow uploads for testing/fallback (dev rule)
    }
    
    // Allow public profile pictures
    match /users/{uid}/profile/{allPaths=**} {
      allow read: if true;
      allow write: if true;  // Allow uploads for testing/fallback (dev rule)
    }
    
    // Allow posts storage (including nested media paths like posts/{postId}/photo/...)
    match /posts/{postId}/{allPaths=**} {
      allow read: if true;
      // Require authenticated users to write post media
      allow write: if request.auth != null;
    }
    
    // Allow any authenticated user to write to users folder for posts
    match /users/{allPaths=**} {
      allow read: if true;
      allow write: if true;  // Allow writes for dev/fallback
    }
    
    // Default deny all other paths
    match /{allPaths=**} {
      allow read: if false;
      allow write: if false;
    }
  }
}
