// Firebase Storage Security Rules for Wificontent App
// Apply these rules in Firebase Console -> Storage -> Rules tab

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    
    // Allow users to upload their own photos to their network folders
    match /users/{uid}/photos/{allPaths=**} {
      allow read: if true;  // Anyone can view photos
      allow write: if true;  // Allow uploads for testing/fallback (dev rule)
    }
    
    // Allow public profile pictures
    match /users/{uid}/profile/{allPaths=**} {
      allow read: if true;
      allow write: if true;  // Allow uploads for testing/fallback (dev rule)
    }
    
    // Allow posts storage (including nested media paths like posts/{postId}/photo/...)
    match /posts/{postId}/{allPaths=**} {
      allow read: if true;
      // Require authenticated users to write post media
      // Allow common image and video content types with a 25MB size limit for uploads
      allow write: if request.auth != null
                   && request.resource.size < 25 * 1024 * 1024
                   && (
                        request.resource.contentType.matches('image/.*') ||
                        request.resource.contentType.matches('video/.*') ||
                        request.resource.contentType.matches('application/octet-stream')
                      );
    }
    
    // Allow any authenticated user to write to users folder for posts (profile pics & photos)
    match /users/{uid}/{allPaths=**} {
      allow read: if true;
      // Photos and profile images: enforce content type and size limits
      allow write: if request.auth != null && request.auth.uid == uid
                   && request.resource.size < 10 * 1024 * 1024
                   && request.resource.contentType.matches('image/.*');
    }
    
    // Default deny all other paths
    match /{allPaths=**} {
      allow read: if false;
      allow write: if false;
    }
  }
}
